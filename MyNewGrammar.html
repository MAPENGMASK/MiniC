NON-TERMINALS

Start ::= program <EOF>
program ::= ( <INT> | <VOID> ) <MAIN> <LC> <RC> <LB> ( Statementblock )* <RB>
Statementblock ::= Statement
| <LB> ( Statementblock )* <RB>
Statement ::= ( SequenceStatement | Conditionalstatements | loopstatement | SwitchStatement | Return )
SequenceStatement ::= ( statement | assignment )? <SEMICOLON>
statement ::= type <IDENTIFIER> ( <ASSIGN> ( <INTEGER_LITERAL> | <FLOAT_LITERAL> ) )? ( <COMMA> <IDENTIFIER> ( <ASSIGN> ( <INTEGER_LITERAL> | <FLOAT_LITERAL> ) )? )*
assignment ::= <IDENTIFIER> ( <ADDONE> | <MINUTEONE> )? <ASSIGN> Expression ( <COMMA> <IDENTIFIER> <ASSIGN> Expression )*
Conditionalstatements ::= <IF> <LC> Logiccondition <RC> Statementblock ( <ELSE> Statementblock )?
Logiccondition ::= Condition ( Logic Logiccondition )?
Condition ::= Expression ( Relationship Expression )?
loopstatement ::= ( <WHILE> <LC> Logiccondition <RC> Statementblock )
| ( <DO> Statementblock <WHILE> <LC> Logiccondition <RC> <SEMICOLON> )
| ( <FOR> <LC> statement <SEMICOLON> Logiccondition <SEMICOLON> crease <RC> Statementblock )
SwitchStatement ::= <SWITCH> <LC> Expression <RC> <LB> ( <CASE> <INTEGER_LITERAL> <COLON> Statementblock ( <BREAK> <SEMICOLON> )? )+ ( <DEF> <COLON> Statementblock ( <BREAK> <SEMICOLON> )? )? <RB>
Expression ::= AdditiveExpression
AdditiveExpression ::= MultiplicativeExpression ( ( <ADD> | <MIN> ) MultiplicativeExpression )*
MultiplicativeExpression ::= UnaryExpression ( ( <MUL> | <DIV> | <QUEUE> ) UnaryExpression )*
UnaryExpression ::= <LC> Expression <RC>
| <INTEGER_LITERAL>
| <FLOAT_LITERAL>
| <IDENTIFIER>
type ::= ( <INT> | <STRING> | <DOUBLE> | <FLOAT> | <CHAR> )
Relationship ::= ( <GT> | <LT> | <GE> | <LE> | <NE> | <EQ> )
Logic ::= <AND>
| <OR>
Return ::= <RETURN> ( <INTEGER_LITERAL> | <FLOAT_LITERAL> )? <SEMICOLON>
crease ::= <IDENTIFIER> ( <ADDONE> | <MINUTEONE> )?
